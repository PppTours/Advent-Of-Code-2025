Sub Part1()

    Dim dial As Integer
    Dim count As Integer
    Dim i As Integer
    Dim cell As String
    dial = 50
    count = 0
    i = 1
    
    While Not IsEmpty(Application.ActiveSheet.Range("A" & i).Value)
        Let cell = Application.ActiveSheet.Range("A" & i).Value
        If StrComp(Left(cell, 1), "R") = 0 Then
            dial = dial + CInt(Right(cell, Len(cell) - 1))
            While dial > 99
                dial = dial - 100
            Wend
        Else
            dial = dial - CInt(Right(cell, Len(cell) - 1))
            While dial < 0
                dial = dial + 100
            Wend
        End If
        
        If dial = 0 Then
            count = count + 1
        End If
        i = i + 1
    Wend
    
    Application.ActiveSheet.Range("B1").Value = CStr(count)
End Sub
